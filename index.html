    class Student {
    
    // encapsulation \/ all private
    private String Name;
    private double[] Grades;

    // no need for a constructor as the invoked constructor in main is an
    // empty-parameter constructor that java already takes care of creating it
    
    // encapsulation \/ setters and getters
    public String getName() {
        return Name;
    }

    public void setName(String Name) {
        this.Name = Name;
    }

    public double[] getGrades() {
        return Grades;
    }

    public void setGrades(double[] Grades) {
        this.Grades = Grades;
    }
    
    // abstraction \/ all class work is done within the class
    public int checkGrades() {
        int counter = 0;

        for (double x : Grades) {

            if (x < 60) {
                counter++;
            }

        }

        return counter;
    }

}

public class TestingStudent {

    public static void main(String args[]) {
        
        Student e1 = new Student();
        
        // encapsulation \/ setters
        e1.setName("Manal Kaled");
        
        e1.setGrades(new double[]{85, 75, 45, 60});
        
        // encapsulation \/ getters
        System.out.print("The Student Name is: " + e1.getName());
        
        // abstraction \/ checkGrades method within its class
        System.out.println(" Failed in: " + e1.checkGrades() + " courses");
    }
    
    // abstraction \/ method deleted
    // because we want to separate class implementation from the use of the class
    
}

class Person {

        Name = name;

        Address = address;

        Siblings = siblings;

        return Name;

    class Polygon {

    private String polygonName;
    private int polygonSides;
    private double exteriorAngles;

    public Polygon(String polygonName, int polygonSides) {
        this.polygonName = polygonName;
        this.polygonSides = polygonSides;
    }

    // getters
    public String getPolygonName() {
        return polygonName;
    }

    public int getPolygonSides() {
        return polygonSides;
    }

    public double getExteriorAngles() {
        return exteriorAngles;
    }

    // service or instance method
    public void calcPolygonExteriorAngles() {
        exteriorAngles = (polygonSides - 2) * 180;
    }

    // service or instance method
    @Override
    public String toString() {
        return "Polygon Name: " + polygonName + ", Polygon Sides: " + polygonSides + ", Exterior Angles: " + exteriorAngles;
    }

}
